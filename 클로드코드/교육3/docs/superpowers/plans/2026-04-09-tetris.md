```javascript
// ===== PIECES =====
function rotate(matrix) {
  const rows = matrix.length;
  const cols = matrix[0].length;
  const result = Array.from({ length: cols }, () => Array(rows).fill(0));
  for (let r = 0; r < rows; r++)
    for (let c = 0; c < cols; c++)
      result[c][rows - 1 - r] = matrix[r][c];
  return result;
}

const PIECES = [
  { shape: [[1, 1, 1, 1]],             color: '#29B6F6' }, // I
  { shape: [[1, 1], [1, 1]],           color: '#FFCA28' }, // O
  { shape: [[0, 1, 0], [1, 1, 1]],     color: '#AB47BC' }, // T
  { shape: [[0, 1, 1], [1, 1, 0]],     color: '#66BB6A' }, // S
  { shape: [[1, 1, 0], [0, 1, 1]],     color: '#EF5350' }, // Z
  { shape: [[1, 0, 0], [1, 1, 1]],     color: '#1565C0' }, // J
  { shape: [[0, 0, 1], [1, 1, 1]],     color: '#FF7043' }, // L
];
```

## Task 3: Board 모듈

**Files:**
- Modify: `index.html` (script 섹션 내 `// ===== BOARD =====`)

- [ ] **Step 1: Board 코드 작성**

`// ===== BOARD =====` 주석을 아래 코드로 교체한다.

```javascript
// ===== BOARD =====
const Board = {
  COLS: 10,
  ROWS: 20,
  grid: [],

  init() {
    this.grid = Array.from({ length: this.ROWS }, () => Array(this.COLS).fill(null));
  },

  // shape: 2D 배열, x/y: 보드 좌표 기준 피스 좌상단 위치
  // 반환: 유효하면 true (벽/바닥/기존 블록과 충돌 없음)
  isValid(shape, x, y) {
    for (let r = 0; r < shape.length; r++) {
      for (let c = 0; c < shape[r].length; c++) {
        if (!shape[r][c]) continue;
        const nr = y + r;
        const nc = x + c;
        if (nc < 0 || nc >= this.COLS || nr >= this.ROWS) return false;
        if (nr >= 0 && this.grid[nr][nc] !== null) return false;
      }
    }
    return true;
  },

  // 피스를 그리드에 고정
  place(shape, x, y, color) {
    for (let r = 0; r < shape.length; r++)
      for (let c = 0; c < shape[r].length; c++)
        if (shape[r][c]) this.grid[y + r][x + c] = color;
  },

  // 가득 찬 줄 제거, 제거된 줄 수 반환
  clearLines() {
    let cleared = 0;
    for (let r = this.ROWS - 1; r >= 0; r--) {
      if (this.grid[r].every(cell => cell !== null)) {
        this.grid.splice(r, 1);
        this.grid.unshift(Array(this.COLS).fill(null));
        cleared++;
        r++; // 같은 인덱스 재확인
      }
    }
    return cleared;
  },
};
```

- [ ] **Step 1: Piece 코드 작성**

`// ===== PIECE =====` 주석을 아래 코드로 교체한다.

```javascript
// ===== PIECE =====
const Piece = {
  shape: null,
  color: null,
  pieceIdx: -1,
  x: 0,
  y: 0,

  // idx: PIECES 배열 인덱스
  spawn(idx) {
    this.pieceIdx = idx;
    this.shape = PIECES[idx].shape.map(row => [...row]);
    this.color = PIECES[idx].color;
    this.x = Math.floor((Board.COLS - this.shape[0].length) / 2);
    this.y = 0;
  },

  // 시계방향 90도 회전 (충돌 시 취소)
  rotate() {
    const newShape = rotate(this.shape);
    if (Board.isValid(newShape, this.x, this.y)) {
      this.shape = newShape;
    }
  },

  // dx/dy 방향으로 1칸 이동, 성공 시 true 반환
  move(dx, dy) {
    if (Board.isValid(this.shape, this.x + dx, this.y + dy)) {
      this.x += dx;
      this.y += dy;
      return true;
    }
    return false;
  },

  // 하드 드롭 위치 Y 좌표 계산
  getGhostY() {
    let gy = this.y;
    while (Board.isValid(this.shape, this.x, gy + 1)) gy++;
    return gy;
  },
};
```

**Files:**

- [ ] **Step 1: Renderer 코드 작성**

`// ===== RENDERER =====` 주석을 아래 코드로 교체한다.

```javascript
// ===== RENDERER =====
const Renderer = {
  CELL: 30,
  boardCanvas: null, boardCtx: null,
  nextCanvas: null,  nextCtx: null,
  holdCanvas: null,  holdCtx: null,

  init() {
    this.boardCanvas = document.getElementById('board');
    this.boardCtx    = this.boardCanvas.getContext('2d');
    this.nextCanvas  = document.getElementById('next');
    this.nextCtx     = this.nextCanvas.getContext('2d');
    this.holdCanvas  = document.getElementById('hold');
    this.holdCtx     = this.holdCanvas.getContext('2d');
  },

  // 단일 셀 그리기 (alpha: 고스트용 투명도)
  drawCell(ctx, x, y, color, alpha = 1) {
    const s = this.CELL;
    ctx.globalAlpha = alpha;
    ctx.fillStyle = color;
    ctx.fillRect(x * s + 1, y * s + 1, s - 2, s - 2);
    // 상단 하이라이트
    ctx.fillStyle = 'rgba(255,255,255,0.28)';
    ctx.fillRect(x * s + 2, y * s + 2, s - 4, 5);
    ctx.globalAlpha = 1;
  },

  // 보드 + 피스 + 고스트 + 오버레이 렌더링
  drawBoard() {
    const ctx = this.boardCtx;
    const { COLS, ROWS, grid } = Board;
    const W = this.boardCanvas.width;
    const H = this.boardCanvas.height;

    // 배경
    ctx.fillStyle = '#FAFAFA';
    ctx.fillRect(0, 0, W, H);

    // 그리드 선
    ctx.strokeStyle = '#E0E0E0';
    ctx.lineWidth = 0.5;
    for (let r = 0; r <= ROWS; r++) {
      ctx.beginPath();
      ctx.moveTo(0, r * this.CELL);
      ctx.lineTo(W, r * this.CELL);
      ctx.stroke();
    }
    for (let c = 0; c <= COLS; c++) {
      ctx.beginPath();
      ctx.moveTo(c * this.CELL, 0);
      ctx.lineTo(c * this.CELL, H);
      ctx.stroke();
    }

    // 고정된 블록
    for (let r = 0; r < ROWS; r++)
      for (let c = 0; c < COLS; c++)
        if (grid[r][c]) this.drawCell(ctx, c, r, grid[r][c]);

    // 진행 중일 때: 고스트 + 현재 피스
    if (Game.state === 'playing' || Game.state === 'paused') {
      const gy = Piece.getGhostY();
      for (let r = 0; r < Piece.shape.length; r++)
        for (let c = 0; c < Piece.shape[r].length; c++)
          if (Piece.shape[r][c]) this.drawCell(ctx, Piece.x + c, gy + r, Piece.color, 0.15);

      for (let r = 0; r < Piece.shape.length; r++)
        for (let c = 0; c < Piece.shape[r].length; c++)
          if (Piece.shape[r][c]) this.drawCell(ctx, Piece.x + c, Piece.y + r, Piece.color);
    }

    // 일시정지 오버레이
    if (Game.state === 'paused') {
      ctx.fillStyle = 'rgba(250,250,250,0.78)';
      ctx.fillRect(0, 0, W, H);
      ctx.fillStyle = '#212121';
      ctx.font = 'bold 28px sans-serif';
      ctx.textAlign = 'center';
      ctx.fillText('PAUSED', W / 2, 275);
      ctx.font = '13px sans-serif';
      ctx.fillStyle = '#9E9E9E';
      ctx.fillText('P 키로 재개', W / 2, 308);
    }

    // 게임오버 오버레이
    if (Game.state === 'gameover') {
      ctx.fillStyle = 'rgba(250,250,250,0.88)';
      ctx.fillRect(0, 0, W, H);
      ctx.textAlign = 'center';
      ctx.fillStyle = '#212121';
      ctx.font = 'bold 30px sans-serif';
      ctx.fillText('GAME OVER', W / 2, 230);
      ctx.font = '20px sans-serif';
      ctx.fillText(Game.score.toLocaleString() + ' 점', W / 2, 272);
      ctx.font = '13px sans-serif';
      ctx.fillStyle = '#9E9E9E';
      ctx.fillText('Enter 키로 다시 시작', W / 2, 308);
    }
  },

  // Hold / Next 미니 캔버스에 피스 그리기
  drawMini(ctx, canvas, pieceIdx) {
    ctx.fillStyle = '#FAFAFA';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    if (pieceIdx === null || pieceIdx < 0) return;

    const piece = PIECES[pieceIdx];
    const s = 22;
    const ox = Math.floor((canvas.width  - piece.shape[0].length * s) / 2);
    const oy = Math.floor((canvas.height - piece.shape.length    * s) / 2);

    for (let r = 0; r < piece.shape.length; r++) {
      for (let c = 0; c < piece.shape[r].length; c++) {
        if (piece.shape[r][c]) {
          ctx.fillStyle = piece.color;
          ctx.fillRect(ox + c * s + 1, oy + r * s + 1, s - 2, s - 2);
          ctx.fillStyle = 'rgba(255,255,255,0.28)';
          ctx.fillRect(ox + c * s + 2, oy + r * s + 2, s - 4, 5);
        }
      }
    }
  },

  // 전체 화면 갱신 (매 프레임 호출)
  draw() {
    this.drawBoard();
    this.drawMini(this.nextCtx, this.nextCanvas, Game.nextPieceIdx);
    this.drawMini(this.holdCtx, this.holdCanvas, Game.heldPieceIdx);
    document.getElementById('score').textContent = Game.score.toLocaleString();
    document.getElementById('level').textContent = Game.level;
    document.getElementById('best').textContent  = Game.bestScore.toLocaleString();
  },
};
```

## Task 6: Game 모듈

**Files:**
- Modify: `index.html` (script 섹션 내 `// ===== GAME =====`)

- [ ] **Step 1: Game 코드 작성**

`// ===== GAME =====` 주석을 아래 코드로 교체한다.

```javascript
// ===== GAME =====
const Game = {
  state: 'idle',          // 'idle' | 'playing' | 'paused' | 'gameover'
  score: 0,
  level: 1,
  lines: 0,
  bestScore: 0,
  lastTime: 0,
  dropAccumulator: 0,
  nextPieceIdx: null,
  heldPieceIdx: null,
  canHold: true,
  animId: null,

  // 레벨별 드롭 간격(ms): 레벨1=1000ms, 최소 100ms
  getDropInterval() {
    return Math.max(100, 1000 - (this.level - 1) * 80);
  },

  randomIdx() {
    return Math.floor(Math.random() * PIECES.length);
  },

  start() {
    this.score = 0;
    this.level = 1;
    this.lines = 0;
    this.canHold = true;
    this.heldPieceIdx = null;
    this.lastTime = 0;
    this.dropAccumulator = 0;

    try {
      this.bestScore = parseInt(localStorage.getItem('tetris_best_score')) || 0;
    } catch (e) {
      this.bestScore = 0;
    }

    Board.init();
    this.nextPieceIdx = this.randomIdx();
    this.spawnPiece();

    if (this.state !== 'gameover') {
      this.state = 'playing';
      if (this.animId) cancelAnimationFrame(this.animId);
      this.animId = requestAnimationFrame(ts => this.loop(ts));
    }
  },

  spawnPiece() {
    Piece.spawn(this.nextPieceIdx);
    this.nextPieceIdx = this.randomIdx();
    this.canHold = true;

    // 스폰 위치 즉시 충돌 → 게임오버
    if (!Board.isValid(Piece.shape, Piece.x, Piece.y)) {
      this.gameOver();
    }
  },

  loop(ts) {
    if (this.state !== 'playing') return;

    const delta = this.lastTime ? ts - this.lastTime : 0;
    this.lastTime = ts;
    this.dropAccumulator += delta;

    if (this.dropAccumulator >= this.getDropInterval()) {
      this.dropAccumulator = 0;
      if (!Piece.move(0, 1)) this.lockPiece();
    }

    Renderer.draw();
    this.animId = requestAnimationFrame(ts => this.loop(ts));
  },

  lockPiece() {
    Board.place(Piece.shape, Piece.x, Piece.y, Piece.color);
    const cleared = Board.clearLines();
    if (cleared > 0) this.addLineScore(cleared);
    this.spawnPiece();
  },

  // 줄 클리어 점수 + 레벨업
  addLineScore(cleared) {
    const pts = [0, 100, 300, 500, 800];
    this.score += pts[cleared] * this.level;
    this.lines += cleared;
    this.level = Math.floor(this.lines / 10) + 1;
  },

  softDrop() {
    if (Piece.move(0, 1)) {
      this.score += 1;
      this.dropAccumulator = 0;
    }
  },

  hardDrop() {
    let count = 0;
    while (Piece.move(0, 1)) count++;
    this.score += count * 2;
    this.lockPiece();
    this.dropAccumulator = 0;
  },

  hold() {
    if (!this.canHold) return;
    this.canHold = false;

    if (this.heldPieceIdx === null) {
      // 처음 홀드: 현재 피스 저장 후 새 피스 스폰
      this.heldPieceIdx = Piece.pieceIdx;
      this.spawnPiece();
    } else {
      // 교체: 홀드 슬롯 ↔ 현재 피스
      const temp = this.heldPieceIdx;
      this.heldPieceIdx = Piece.pieceIdx;
      Piece.spawn(temp);
    }
  },

  gameOver() {
    this.state = 'gameover';
    try {
      if (this.score > this.bestScore) {
        this.bestScore = this.score;
        localStorage.setItem('tetris_best_score', String(this.bestScore));
      }
    } catch (e) { /* localStorage 미지원 */ }
    Renderer.draw();
  },

  pause() {
    if (this.state === 'playing') {
      this.state = 'paused';
      Renderer.draw();
    } else if (this.state === 'paused') {
      this.state = 'playing';
      this.lastTime = 0;
      this.animId = requestAnimationFrame(ts => this.loop(ts));
    }
  },
};
```

```javascript
// ===== INPUT + INIT =====
const Input = {
  init() {
    document.addEventListener('keydown', e => {
      // 게임오버 상태: Enter로 재시작
      if (Game.state === 'gameover') {
        if (e.code === 'Enter') {
          e.preventDefault();
          Game.start();
        }
        return;
      }

      // 일시정지 토글 (어떤 상태에서든)
      if (e.code === 'KeyP') {
        Game.pause();
        return;
      }

      // 게임 진행 중에만 동작
      if (Game.state !== 'playing') return;

      switch (e.code) {
        case 'ArrowLeft':            Piece.move(-1, 0); break;
        case 'ArrowRight':           Piece.move(1, 0);  break;
        case 'ArrowDown':            Game.softDrop();   break;
        case 'ArrowUp': case 'KeyZ': Piece.rotate();    break;
        case 'Space':                Game.hardDrop();   break;
        case 'KeyC':                 Game.hold();       break;
        default: return; // 해당 없는 키는 preventDefault 생략
      }

      e.preventDefault();
      Renderer.draw();
    });
  },
};

window.onload = () => {
  Renderer.init();
  Input.init();
  Game.start();
};
```
